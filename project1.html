def year_country_pop(country_row):
    '''Gets the population of a country for a year from a row'''
    country_pop = population.where('time', country_row.item('time')).where('geo', country_row.item('geo')).column('population_total').item(0)
    return country_pop

def poverty_timeline(country):
    '''Draw a timeline of people living in extreme poverty in a country.'''
    geo = countries.where('name', country).column('country').item(0)
    poverty_in_country = poverty.where('geo', geo)
    
    table = Table().with_columns('Year', poverty_in_country.column('time'), 'Number in Poverty', poverty_in_country.column('extreme_poverty_percent_people_below_125_a_day') /100 * poverty_in_country.apply(year_country_pop))
    table.plot('Year', 'Number in Poverty')
